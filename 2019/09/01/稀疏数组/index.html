# 介绍：

在二维数组种，如果存在大量的空位（默认为0或者其他）。会比较浪费内存，可以通过一种数据结构（稀疏数组）来表示二维数组，但是不会浪费这么多的内存。 讲白了就是存下数组中有用的索引和值，压缩数据。

# Show Code:

原数组为一个11*11的二位数组

public static void main(String[] args) {
        int[][] originArr=new int[11][11];
        //为空数组赋值有效元素
        originArr[1][2]=1;
        originArr[2][3]=2;
        originArr[3][4]=2;
        int sum=0;//非空元素的数量
        for (int[] row:originArr){
            System.out.println();
            for (int data:row){
//                System.out.print(data+"\t");
                if (data!=0) sum++;
            }
        }

        int sparseArr[][]=new int[sum+1][3];//创建稀疏数组
        //二维数组中第一个元素寄存原数组二维模型
        sparseArr[0][0]=11;
        sparseArr[0][1]=11;
        sparseArr[0][2]=sum;
        int count=0;//索引的计数器
        for (int i = 0; i < 11; i++) {
            for (int j = 0; j < 11; j++) {
                if (originArr[i][j]!=0){
                    sparseArr[++count][0]=i;
                    sparseArr[count][1]=j;
                    sparseArr[count][2]=originArr[i][j];
                }
            }
        }

    //    for (int [] row:sparseArr){//打印看看稀疏数组
//            System.out.println();
//            System.out.println(row[0]+" "+row[1]+" "+row[2]);
//        }

        int[][] restoreArr=new int[sparseArr[0][0]][sparseArr[0][1]];
        for (int i = 1; i < sparseArr.length; i++) {//从稀疏数组恢复原数组
                restoreArr[sparseArr[i][0]][sparseArr[i][1]]=sparseArr[i][2];
        }
        for (int [] row:restoreArr){//打印一下原数组
            System.out.println();
            for (int data:row){
                System.out.print(data+"\t");
                if (data!=0) sum++;
            }
        }
    }
